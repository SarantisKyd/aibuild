function showPage(id) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-links a').forEach(a => a.classList.remove('active'));
  document.getElementById('page-'+id).classList.add('active');
  document.getElementById('nav-'+id).classList.add('active');
  if (id === 'board') renderJobs(JOBS);
  if (id === 'marketplace') renderTools();
  window.scrollTo(0, 0);
}

function renderJobs(jobs) {
  const list = document.getElementById('jobs-list');
  list.innerHTML = jobs.map(j => `
    <div class="job-card" onclick="openJob(${j.id})">
      <div class="job-top">
        <div class="job-title">${j.title}</div>
        <div class="job-budget">$${j.budget}</div>
      </div>
      <div class="job-desc">${j.desc.substring(0,120)}…</div>
      <div class="job-meta">
        ${j.urgent ? '<span class="tag urgent">⚡ Urgent</span>' : ''}
        ${j.isNew ? '<span class="tag new">✦ New</span>' : ''}
        <span class="tag">🕐 ${j.deadline}</span>
        ${j.skills.slice(0,3).map(s => `<span class="tag">${s}</span>`).join('')}
        <span class="bid-count">${j.bids} bids</span>
      </div>
    </div>
  `).join('');
}

function filterJobs(el, cat) {
  document.querySelectorAll('.filter-chip').forEach(c => c.classList.remove('active'));
  el.classList.add('active');
  const filtered = cat === 'all' ? JOBS : JOBS.filter(j => j.cat === cat);
  renderJobs(filtered);
}

function submitBid() {

function closeModal(e) {
  if (e.target === document.getElementById('job-modal')) {
    document.getElementById('job-modal').classList.remove('open');
  }
}

function toggleSkill(el) { el.classList.toggle('selected'); }

function submitJob() {
  const title = document.getElementById('job-title').value.trim();
  const budget = document.getElementById('job-budget').value;
  if (!title) { toast('Add a job title first'); return; }
  if (!budget) { toast('Add a budget'); return; }
  toast('✓ Job posted! Builders will bid within hours.');
  setTimeout(() => showPage('board'), 1500);
}

function toast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 3000);
}
